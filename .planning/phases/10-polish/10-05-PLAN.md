1. **Component structure**:
```typescript
'use client';

import { useState } from 'react';
import { useWindowManager } from '@/app/contexts/WindowManagerContext';
import type { ThemeName } from '@/app/lib/themes';

interface ContextMenuProps {
  x: number;
  y: number;
  onClose: () => void;
}

export function ContextMenu({ x, y, onClose }: ContextMenuProps) {
  const { theme } = useWindowManager();
  const [showAppearanceSubmenu, setShowAppearanceSubmenu] = useState(false);
  
  const handleThemeSelect = (themeName: ThemeName) => {
    theme.setTheme(themeName);
    onClose();
  };
  
  return (
    <div 
      className="context-menu win98-menu"
      style={{ position: 'fixed', left: x, top: y, zIndex: 10000 }}
    >
      <div className="menu-item" onClick={onClose}>
        <span>Refresh</span>
      </div>
      <div className="menu-separator" />
      <div 
        className="menu-item"
        onMouseEnter={() => setShowAppearanceSubmenu(true)}
        onMouseLeave={() => setShowAppearanceSubmenu(false)}
      >
        <span>Appearance</span>
        <span className="menu-arrow">▶</span>
        
        {showAppearanceSubmenu && (
          <div className="submenu">
            {theme.availableThemes.map((t) => (
              <div
                key={t.name}
                className="menu-item"
                onClick={() => handleThemeSelect(t.name)}
              >
                <span className="radio-indicator">
                  {theme.currentTheme === t.name ? '●' : '○'}
                </span>
                <span>{t.displayName}</span>
              </div>
            ))}
          </div>
        )}
      </div>
      <div className="menu-separator" />
      <div className="menu-item" onClick={onClose}>
        <span>Properties</span>
      </div>
    </div>
  );
}
```

3. **Add right-click handler** to desktop div:
```typescript
const handleContextMenu = (e: React.MouseEvent) => {
  e.preventDefault();
  // Only show if clicking desktop itself, not icons or windows
  if (e.target === e.currentTarget) {
    setContextMenu({ x: e.clientX, y: e.clientY });
  }
};

const handleCloseContextMenu = () => {
  setContextMenu(null);
};

// In JSX:
<div 
  className="desktop"
  onContextMenu={handleContextMenu}
  onClick={handleCloseContextMenu} // Close on any click
>
  {/* ... existing desktop content */}
  
  {contextMenu && (
    <ContextMenu 
      x={contextMenu.x} 
      y={contextMenu.y} 
      onClose={handleCloseContextMenu}
    />
  )}
</div>
```

4. **Close context menu on Escape key**:
```typescript
useEffect(() => {
  const handleEscape = (e: KeyboardEvent) => {
    if (e.key === 'Escape' && contextMenu) {
      setContextMenu(null);
    }
  };
  
  window.addEventListener('keydown', handleEscape);
  return () => window.removeEventListener('keydown', handleEscape);
}, [contextMenu]);
```
